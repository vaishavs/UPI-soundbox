#include <SoftwareSerial.h>

 

SoftwareSerial gsmSerial(16, 17); // RX, TX (ESP32 pins)

 

void setup() {

  Serial.begin(115200);

  gsmSerial.begin(9600); // Set the baud rate of your GSM module

  delay(1000);

  Serial.println("Initializing GSM module...");

  delay(2000); // Wait for the GSM module to initialize

}

 

void loop() {

  if (gsmSerial.available()) {

    char c = gsmSerial.read();

    Serial.print(c);

  }

 

  if (Serial.available()) {

    char c = Serial.read();

    gsmSerial.print(c);

  }

}
